Exercise 1 - Basic Git Workflow

 mkdir basic-git
 cd basic-git
 git init

 touch README.md
 git add README.md
 git commit -m "Initial commit with README"

 echo "This is my first Git project" >> README.md
 git add README.md
 git commit -m "Updated README file"

 git log


Exercise 2 - Branching and Merging

 git branch feature
 git checkout feature
 touch feature.txt
 git add feature.txt
 git commit -m "Added feature file"
 
 git checkout main
 git merge feature
 git branch -d feature


Exercise 3 - Conflict Resolution
 
 git checkout main
 git branch branch1
 git branch branch2

 git checkout branch1
 echo "Hello from branch1" > conflict.txt
 git add conflict.txt
 git commit -m "Branch1 change"

 git checkout main
 git merge branch1
 git merge branch2

 nano conflict.txt
 git add conflict.txt
 git commit -m "Resolved merge conflict"


Exercise 4 - GitHub Collaboration

 git clone https://github.com/Swpnaneel/repository-name.git
 cd repository-name

 git checkout -b my-feature
 touch newfile.txt
 git add newfile.txt
 git commit -m "Added new feature"
 git push origin my-feature


Exercise 5 - Advanced Workflow

 mkdir advanced-git
 cd advanced-git
 git init
 touch file1.txt file2.txt
 git add .
 git commit -m "Added multiple files"

 touch .gitignore
 echo "node_modules/" >> .gitignore
 git add .gitignore
 git commit -m "Added gitignore"
 
 echo "Temporary work" >> file1.txt
 git stash
 git tag -a v1.0 -m "First release"

 git remote add origin https://github.com/Swapnaneel/advanced-git.git
 git push origin main
 git push origin --tags
